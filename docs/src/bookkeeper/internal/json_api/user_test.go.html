   334  func requireBodyMatchUser(t *testing.T, body *bytes.Buffer, user db.User) {
   335  	data, err := io.ReadAll(body)
   336  	require.NoError(t, err)
   337  
   338  	var gotUser db.User
   339  	err = json.Unmarshal(data, &gotUser)
   340  
   341  	require.NoError(t, err)
   342  	require.Equal(t, user.Username, gotUser.Username)
   343  	require.Equal(t, user.FullName, gotUser.FullName)
   344  	require.Equal(t, user.Email, gotUser.Email)
   345  	require.Empty(t, gotUser.HashedPassword)
